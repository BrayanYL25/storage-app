src/
│
├── core/                   # Módulos y lógica central compartida entre funcionalidades
│   ├── services/           # Servicios o módulos comunes (APIs, autenticación, etc.)
│   ├── utils/              # Utilidades generales (helpers, formateadores, validaciones)
│   └── hooks/              # Hooks reutilizables en toda la aplicación
│
├── features/               # Funcionalidades principales del sistema
│   ├── auth/               # Funcionalidad de autenticación
│   │   ├── components/     # Componentes específicos de auth (Ej: Login, Register)
│   │   ├── hooks/          # Hooks específicos de auth
│   │   ├── pages/          # Páginas o vistas específicas de auth
│   │   ├── services/       # Servicios específicos de auth (Ej: AuthService)
│   │   └── styles/         # Estilos específicos de auth
│   │
│   ├── users/              # Funcionalidad de gestión de usuarios
│   │   ├── components/     # Componentes específicos de usuarios
│   │   ├── hooks/          # Hooks específicos de usuarios
│   │   ├── pages/          # Páginas o vistas específicas de usuarios
│   │   ├── services/       # Servicios específicos de usuarios (Ej: UserService)
│   │   └── styles/         # Estilos específicos de usuarios
│   │
│   ├── products/           # Funcionalidad de gestión de productos
│   │   ├── components/     # Componentes específicos de productos
│   │   ├── hooks/          # Hooks específicos de productos
│   │   ├── pages/          # Páginas o vistas específicas de productos
│   │   ├── services/       # Servicios específicos de productos
│   │   └── styles/         # Estilos específicos de productos
│   │
│   └── .../                # Otras funcionalidades clave del sistema
│
├── shared/                 # Módulos y recursos compartidos entre todas las funcionalidades
│   ├── components/         # Componentes UI reutilizables (Botones, Formularios, etc.)
│   ├── hooks/              # Hooks reutilizables
│   ├── constants/          # Constantes y configuraciones compartidas
│   ├── styles/             # Estilos compartidos (CSS, SASS)
│   └── utils/              # Utilidades generales compartidas
│
├── assets/                 # Recursos estáticos como imágenes, fuentes, etc.
│
└── app/                    # Configuración general de la aplicación
    ├── routes/             # Configuración de rutas
    ├── context/            # Contextos de React
    └── store/              # Estado global (si usas una librería de manejo de estado como Redux)
